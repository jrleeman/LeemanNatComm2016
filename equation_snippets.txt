% Equation - Rate and State Friction %
\begin{equation}
	\mu =  a  \ln\left(\frac{V}{V_0}\right) + b \ln\left(\frac{V_0 \theta}
    {D_c}\right) + b_2 \ln\left(\frac{V_0 \theta_2}{D_{c2}}\right)
    \label{eq:rsf}
\end{equation}
% End Equation %

--------------------------------------------------------------------------------

% Equation - Slowness Law %
\begin{equation}
	\frac{\text{d}\theta}{\text{d}t} = 1 - \frac{V \theta}{D_c}
	\label{eq:slowness_law}
\end{equation}
% End Equation %

--------------------------------------------------------------------------------

% Equation - Slip Law %
\begin{equation}
    \frac{\text{d}\theta}{\text{d}t} = -\frac{V \theta}{D_c}\text{ln}
    \frac{V \theta}{D_c}
    \label{eq:slip_law}
\end{equation}
% End Equation %

--------------------------------------------------------------------------------

% Equation - Critical Radius %
\begin{equation}
    r_c = \frac{16}{7\pi}G \frac{D_c}{\sigma_n (b-a)}
    \label{eq:critical_radius}
\end{equation}
% End Equation %

--------------------------------------------------------------------------------

% Equation - Critical Stiffness %

\begin{equation}
    k_c = \frac{b-a}{D_c}
	\label{eq:critical_stiffness}
\end{equation}
% End Equation %
